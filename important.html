$('select').on('change', function(){ $('select option').prop("disabled", false); $("select").not(this).find("option[value="+ $(this).val() + "]").prop('disabled', true); }); var $selects = $('select'); $selects.on('change', function() { // enable all options $selects.find('option').prop('disabled', false); // loop over each select, use its value to // disable the options in the other selects $selects.each(function() { $selects.not(this) .find('option[value="' + this.value + '"]') .prop('disabled', true); }); });
